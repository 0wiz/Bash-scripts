### 1 Permissions
sudo visudo
# Change    "ALL=(ALL:ALL) ALL" to "ALL=(ALL:ALL) NOPASSWD: ALL"
# Add       "admin  ALL=(ALL:ALL) NOPASSWD: ALL"
# Use Ctrl+Shift+O to save and Ctrl+Shift+X to exit visudo

### 2 Clean before update
sudo apt-get -y purge firefox* libreoffice* gnome-calendar* gnome-contacts* gnome-weather* geary*
# If not all of libreoffice is removed then try next line
sudo apt-get -y purge libreoffice-calc libreoffice-draw libreoffice-impress libreoffice-math libreoffice-writer
sudo apt -y autoremove
sudo apt-get clean

### 3 Update
sudo apt -y update
sudo apt -y upgrade
sudo apt -y dist-upgrade
sudo apt -y autoremove
sudo apt-get clean
sudo fwupdmgr get-devices
sudo fwupdmgr get-updates
sudo fwupdmgr -y update
sudo flatpak update
sudo reboot now
# If bloatware has returned to Applications/Pop!_Shop then redo step 2
sudo nautilus
# Show hidden files and delete residuals "firefox", "mozilla", "libreoffice" and "geary"

### 4 Install Brave Browser
sudo apt -y install curl software-properties-common apt-transport-https
curl -s https://brave-browser-apt-release.s3.brave.com/brave-core.asc | sudo apt-key --keyring /etc/apt/trusted.gpg.d/brave-browser-release.gpg add -
echo "deb [arch=amd64] https://brave-browser-apt-release.s3.brave.com/ stable main" | sudo tee /etc/apt/sources.list.d/brave-browser-release.list
sudo apt update
sudo apt install -y brave-browser

### 5 Install GitHub Desktop
# Download https://github.com/shiftkey/desktop/releases/download/release-2.6.3-linux1/GitHubDesktop-linux-2.6.3-linux1.deb
sudo apt-get -y install gdebi-core
sudo gdebi ~/Downloads/GitHubDesktop-linux-2.6.3-linux1.deb

### 6 Setup my shortcuts
# Clone Bash-scripts repo
chmod +x ~/Documents/GitHub/Bash-scripts/Toggle_GNOME-topbar.sh
# Add keyboard shortcut "bash /home/admin/Documents/GitHub/Bash-scripts/Toggle_GNOME-topbar.sh"
# Install dconf Editor and set /org/desktop/wm/keybindings/always-on-top to ['<Super>A']
# To configure gestures, run the next line (Reconfigure preset gestures using keyboard shortcuts)
flatpak install flathub com.github.joseexposito.touche
